#### IMPORTANT!
# This crontab job can only be setup on "crontab -e", NOT "sudo crontab -e"

# Reboot at 7am
0 7 * * * /sbin/shutdown -r +5

# Update IP address after reboot and every hour at 30 minute
#@reboot (sleep 4m; cd /home/pi/project/ || exit 1; /usr/bin/python /home/pi/project/batch_update_ip_to_dropbox.py &)
@reboot sleep 4m; env DISPLAY=:0 lxterminal -e "/bin/bash -c '/usr/bin/python /home/pi/project/batch_update_ip_to_dropbox.py'"
30 * * * * (cd /home/pi/project/ || exit 1; /usr/bin/python /home/pi/project/batch_update_ip_to_dropbox.py &)

# Fetch device settings
@reboot sleep 5m; env DISPLAY=:0 lxterminal -e "/bin/bash -c '/usr/bin/python /home/pi/project/batch_device_settings.py'"

# Fetch media files from server 5 minutes after reboot
#@reboot (sleep 7m; cd /home/pi/project/ || exit 1; /usr/bin/python /home/pi/project/batch_prepare_playlist.py &)
@reboot sleep 7m; env DISPLAY=:0 lxterminal -e "/bin/bash -c '/usr/bin/python /home/pi/project/batch_prepare_playlist.py'"

# Play media files 15 minutes after reboot. The script runs in foreground
#@reboot (sleep 30s; cd /home/pi/project/ || exit 1; DISPLAY=:0 /usr/bin/python /home/pi/project/batch_main.py )
@reboot sleep 15m; env DISPLAY=:0 lxterminal -e "/bin/bash -c '/usr/bin/python /home/pi/project/batch_main.py'"

### Turn on/off the monitor every weekday morning
0,30 * * * * (cd /home/pi/project/ || exit 1; /usr/bin/python /home/pi/project/batch_turn_on_off_hdmi.py &)
